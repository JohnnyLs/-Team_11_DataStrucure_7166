276 // Returns a string of SVG code for an image depicting the given QR Code, with the given number
277 // of border modules. The string always uses Unix newlines (\n), regardless of the platform.
278 static std::string toSvgString(const QrCode& qr, int border) {
279  if (border < 0)
280  throw std::domain_error("Border must be non-negative");
281  if (border > INT_MAX / 2 || border * 2 > INT_MAX - qr.getSize())
282  throw std::overflow_error("Border too large");
283 
284  std::ostringstream sb;
285  sb << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n";
286  sb << "<!DOCTYPE svg PUBLIC \"-//W3C//DTD SVG 1.1//EN\" \"http://www.w3.org/Graphics/SVG/1.1/DTD/svg11.dtd\">\n";
287  sb << "<svg xmlns=\"http://www.w3.org/2000/svg\" version=\"1.1\" viewBox=\"0 0 ";
288  sb << (qr.getSize() + border * 2) << " " << (qr.getSize() + border * 2) << "\" stroke=\"none\">\n";
289  sb << "\t<rect width=\"100%\" height=\"100%\" fill=\"#FFFFFF\"/>\n";
290  sb << "\t<path d=\"";
291  for (int y = 0; y < qr.getSize(); y++) {
292  for (int x = 0; x < qr.getSize(); x++) {
293  if (qr.getModule(x, y)) {
294  if (x != 0 || y != 0)
295  sb << " ";
296  sb << "M" << (x + border) << "," << (y + border) << "h1v1h-1z";
297  }
298  }
299  }
300  sb << "\" fill=\"#000000\"/>\n";
301  sb << "</svg>\n";
302  return sb.str();
303 }
