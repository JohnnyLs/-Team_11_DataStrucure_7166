69  QrSegment QrSegment::makeNumeric(const char* digits) {
70  BitBuffer bb;
71  int accumData = 0;
72  int accumCount = 0;
73  int charCount = 0;
74  for (; *digits != '\0'; digits++, charCount++) {
75  char c = *digits;
76  if (c < '0' || c > '9')
77  throw std::domain_error("String contains non-numeric characters");
78  accumData = accumData * 10 + (c - '0');
79  accumCount++;
80  if (accumCount == 3) {
81  bb.appendBits(static_cast<uint32_t>(accumData), 10);
82  accumData = 0;
83  accumCount = 0;
84  }
85  }
86  if (accumCount > 0) // 1 or 2 digits remaining
87  bb.appendBits(static_cast<uint32_t>(accumData), accumCount * 3 + 1);
88  return QrSegment(Mode::NUMERIC, charCount, std::move(bb));
89  }

92  QrSegment QrSegment::makeAlphanumeric(const char* text) {
93  BitBuffer bb;
94  int accumData = 0;
95  int accumCount = 0;
96  int charCount = 0;
97  for (; *text != '\0'; text++, charCount++) {
98  const char* temp = std::strchr(ALPHANUMERIC_CHARSET, *text);
99  if (temp == nullptr)
100  throw std::domain_error("String contains unencodable characters in alphanumeric mode");
101  accumData = accumData * 45 + static_cast<int>(temp - ALPHANUMERIC_CHARSET);
102  accumCount++;
103  if (accumCount == 2) {
104  bb.appendBits(static_cast<uint32_t>(accumData), 11);
105  accumData = 0;
106  accumCount = 0;
107  }
108  }
109  if (accumCount > 0) // 1 character remaining
110  bb.appendBits(static_cast<uint32_t>(accumData), 6);
111  return QrSegment(Mode::ALPHANUMERIC, charCount, std::move(bb));
112  }

116  // Select the most efficient segment encoding automatically
117  vector<QrSegment> result;
118  if (*text == '\0'); // Leave result empty
119  else if (isNumeric(text))

121  else if (isAlphanumeric(text))

133  QrSegment QrSegment::makeEci(long assignVal) {
134  BitBuffer bb;
135  if (assignVal < 0)
136  throw std::domain_error("ECI assignment value out of range");
137  else if (assignVal < (1 << 7))
138  bb.appendBits(static_cast<uint32_t>(assignVal), 8);
139  else if (assignVal < (1 << 14)) {
140  bb.appendBits(2, 2);
141  bb.appendBits(static_cast<uint32_t>(assignVal), 14);
142  }
143  else if (assignVal < 1000000L) {
144  bb.appendBits(6, 3);
145  bb.appendBits(static_cast<uint32_t>(assignVal), 21);
146  }
147  else
148  throw std::domain_error("ECI assignment value out of range");
149  return QrSegment(Mode::ECI, 0, std::move(bb));
150  }

171  int QrSegment::getTotalBits(const vector<QrSegment>& segs, int version) {
172  int result = 0;
173  for (const QrSegment& seg : segs) {
174  int ccbits = seg.mode->numCharCountBits(version);
175  if (seg.numChars >= (1L << ccbits))
176  return -1; // The segment's length doesn't fit the field's bit width
177  if (4 + ccbits > INT_MAX - result)
178  return -1; // The sum will overflow an int type
179  result += 4 + ccbits;
180  if (seg.data.size() > static_cast<unsigned int>(INT_MAX - result))
181  return -1; // The sum will overflow an int type
182  result += static_cast<int>(seg.data.size());
183  }
184  return result;
185  }

249  QrCode QrCode::encodeSegments(const vector<QrSegment>& segs, Ecc ecl,
250  int minVersion, int maxVersion, int mask, bool boostEcl) {
251  if (!(MIN_VERSION <= minVersion && minVersion <= maxVersion && maxVersion <= MAX_VERSION) || mask < -1 || mask > 7)
252  throw std::invalid_argument("Invalid value");
253 
254  // Find the minimal version number to use
255  int version, dataUsedBits;
256  for (version = minVersion; ; version++) {
257  int dataCapacityBits = getNumDataCodewords(version, ecl) * 8; // Number of data bits available
258  dataUsedBits = QrSegment::getTotalBits(segs, version);
259  if (dataUsedBits != -1 && dataUsedBits <= dataCapacityBits)
260  break; // This version number is found to be suitable
261  if (version >= maxVersion) { // All versions in the range could not fit the given data
262  std::ostringstream sb;
263  if (dataUsedBits == -1)
264  sb << "Segment too long";
265  else {
266  sb << "Data length = " << dataUsedBits << " bits, ";
267  sb << "Max capacity = " << dataCapacityBits << " bits";
268  }
269  throw data_too_long(sb.str());
270  }
271  }
272  if (dataUsedBits == -1)
273  throw std::logic_error("Assertion error");
274 
275  // Increase the error correction level while the data still fits in the current version number
276  for (Ecc newEcl : {Ecc::MEDIUM, Ecc::QUARTILE, Ecc::HIGH}) { // From low to high
277  if (boostEcl && dataUsedBits <= getNumDataCodewords(version, newEcl) * 8)
278  ecl = newEcl;
279  }
280 
281  // Concatenate all segments to create the data bit string
282  BitBuffer bb;
283  for (const QrSegment& seg : segs) {
284  bb.appendBits(static_cast<uint32_t>(seg.getMode().getModeBits()), 4);
285  bb.appendBits(static_cast<uint32_t>(seg.getNumChars()), seg.getMode().numCharCountBits(version));
286  bb.insert(bb.end(), seg.getData().begin(), seg.getData().end());
287  }
288  if (bb.size() != static_cast<unsigned int>(dataUsedBits))
289  throw std::logic_error("Assertion error");
290 
291  // Add terminator and pad up to a byte if applicable
292  size_t dataCapacityBits = static_cast<size_t>(getNumDataCodewords(version, ecl)) * 8;
293  if (bb.size() > dataCapacityBits)
294  throw std::logic_error("Assertion error");
295  bb.appendBits(0, std::min(4, static_cast<int>(dataCapacityBits - bb.size())));
296  bb.appendBits(0, (8 - static_cast<int>(bb.size() % 8)) % 8);
297  if (bb.size() % 8 != 0)
298  throw std::logic_error("Assertion error");
299 
300  // Pad with alternating bytes until data capacity is reached
301  for (uint8_t padByte = 0xEC; bb.size() < dataCapacityBits; padByte ^= 0xEC ^ 0x11)
302  bb.appendBits(padByte, 8);
303 
304  // Pack bits into bytes in big endian
305  vector<uint8_t> dataCodewords(bb.size() / 8);
306  for (size_t i = 0; i < bb.size(); i++)
307  dataCodewords.at(i >> 3) |= (bb.at(i) ? 1 : 0) << (7 - (i & 7));
308 
309  // Create the QR Code object
310  return QrCode(version, ecl, dataCodewords, mask);
311  }

314  QrCode::QrCode(int ver, Ecc ecl, const vector<uint8_t>& dataCodewords, int msk) :
315  // Initialize fields and check arguments
316  version(ver),
317  errorCorrectionLevel(ecl) {
318  if (ver < MIN_VERSION || ver > MAX_VERSION)
319  throw std::domain_error("Version value out of range");
320  if (msk < -1 || msk > 7)
321  throw std::domain_error("Mask value out of range");
322  size = ver * 4 + 17;
323  size_t sz = static_cast<size_t>(size);
324  modules = vector<vector<bool> >(sz, vector<bool>(sz)); // Initially all light
325  isFunction = vector<vector<bool> >(sz, vector<bool>(sz));
326 
327  // Compute ECC, draw modules
328  drawFunctionPatterns();
329  const vector<uint8_t> allCodewords = addEccAndInterleave(dataCodewords);
330  drawCodewords(allCodewords);
331 
332  // Do masking
333  if (msk == -1) { // Automatically choose best mask
334  long minPenalty = LONG_MAX;
335  for (int i = 0; i < 8; i++) {
336  applyMask(i);
337  drawFormatBits(i);
338  long penalty = getPenaltyScore();
339  if (penalty < minPenalty) {
340  msk = i;
341  minPenalty = penalty;
342  }
343  applyMask(i); // Undoes the mask due to XOR
344  }
345  }
346  if (msk < 0 || msk > 7)
347  throw std::logic_error("Assertion error");
348  mask = msk;
349  applyMask(msk); // Apply the final choice of mask
350  drawFormatBits(msk); // Overwrite old format bits
351 
352  isFunction.clear();
353  isFunction.shrink_to_fit();
354  }

382  void QrCode::drawFunctionPatterns() {
383  // Draw horizontal and vertical timing patterns
384  for (int i = 0; i < size; i++) {
385  setFunctionModule(6, i, i % 2 == 0);
386  setFunctionModule(i, 6, i % 2 == 0);
387  }
388 
389  // Draw 3 finder patterns (all corners except bottom right; overwrites some timing modules)
390  drawFinderPattern(3, 3);
391  drawFinderPattern(size - 4, 3);
392  drawFinderPattern(3, size - 4);
393 
394  // Draw numerous alignment patterns
395  const vector<int> alignPatPos = getAlignmentPatternPositions();
396  size_t numAlign = alignPatPos.size();
397  for (size_t i = 0; i < numAlign; i++) {
398  for (size_t j = 0; j < numAlign; j++) {
399  // Don't draw on the three finder corners
400  if (!((i == 0 && j == 0) || (i == 0 && j == numAlign - 1) || (i == numAlign - 1 && j == 0)))
401  drawAlignmentPattern(alignPatPos.at(i), alignPatPos.at(j));
402  }
403  }
404 
405  // Draw configuration data
406  drawFormatBits(0); // Dummy mask value; overwritten later in the constructor
407  drawVersion();

411  void QrCode::drawFormatBits(int msk) {
412  // Calculate error correction code and pack bits
413  int data = getFormatBits(errorCorrectionLevel) << 3 | msk; // errCorrLvl is uint2, msk is uint3
414  int rem = data;
415  for (int i = 0; i < 10; i++)
416  rem = (rem << 1) ^ ((rem >> 9) * 0x537);
417  int bits = (data << 10 | rem) ^ 0x5412; // uint15
418  if (bits >> 15 != 0)
419  throw std::logic_error("Assertion error");
420 
421  // Draw first copy
422  for (int i = 0; i <= 5; i++)
423  setFunctionModule(8, i, getBit(bits, i));
424  setFunctionModule(8, 7, getBit(bits, 6));
425  setFunctionModule(8, 8, getBit(bits, 7));
426  setFunctionModule(7, 8, getBit(bits, 8));
427  for (int i = 9; i < 15; i++)
428  setFunctionModule(14 - i, 8, getBit(bits, i));
429 
430  // Draw second copy
431  for (int i = 0; i < 8; i++)
432  setFunctionModule(size - 1 - i, 8, getBit(bits, i));
433  for (int i = 8; i < 15; i++)
434  setFunctionModule(8, size - 15 + i, getBit(bits, i));
435  setFunctionModule(8, size - 8, true); // Always dark
436  }

439  void QrCode::drawVersion() {
440  if (version < 7)
441  return;
442 
443  // Calculate error correction code and pack bits
444  int rem = version; // version is uint6, in the range [7, 40]
445  for (int i = 0; i < 12; i++)
446  rem = (rem << 1) ^ ((rem >> 11) * 0x1F25);
447  long bits = static_cast<long>(version) << 12 | rem; // uint18
448  if (bits >> 18 != 0)
449  throw std::logic_error("Assertion error");
450 
451  // Draw two copies
452  for (int i = 0; i < 18; i++) {
453  bool bit = getBit(bits, i);
454  int a = size - 11 + i % 3;
455  int b = i / 3;
456  setFunctionModule(a, b, bit);
457  setFunctionModule(b, a, bit);
458  }
459  }

463  for (int dy = -4; dy <= 4; dy++) {
464  for (int dx = -4; dx <= 4; dx++) {

468  setFunctionModule(xx, yy, dist != 2 && dist != 4);

495  vector<uint8_t> QrCode::addEccAndInterleave(const vector<uint8_t>& data) const {
496  if (data.size() != static_cast<unsigned int>(getNumDataCodewords(version, errorCorrectionLevel)))
497  throw std::invalid_argument("Invalid argument");
498 
499  // Calculate parameter numbers
500  int numBlocks = NUM_ERROR_CORRECTION_BLOCKS[static_cast<int>(errorCorrectionLevel)][version];
501  int blockEccLen = ECC_CODEWORDS_PER_BLOCK[static_cast<int>(errorCorrectionLevel)][version];
502  int rawCodewords = getNumRawDataModules(version) / 8;
503  int numShortBlocks = numBlocks - rawCodewords % numBlocks;
504  int shortBlockLen = rawCodewords / numBlocks;
505 
506  // Split data into blocks and append ECC to each block
507  vector<vector<uint8_t> > blocks;
508  const vector<uint8_t> rsDiv = reedSolomonComputeDivisor(blockEccLen);
509  for (int i = 0, k = 0; i < numBlocks; i++) {
510  vector<uint8_t> dat(data.cbegin() + k, data.cbegin() + (k + shortBlockLen - blockEccLen + (i < numShortBlocks ? 0 : 1)));
511  k += static_cast<int>(dat.size());
512  const vector<uint8_t> ecc = reedSolomonComputeRemainder(dat, rsDiv);
513  if (i < numShortBlocks)
514  dat.push_back(0);
515  dat.insert(dat.end(), ecc.cbegin(), ecc.cend());
516  blocks.push_back(std::move(dat));
517  }
518 
519  // Interleave (not concatenate) the bytes from every block into a single sequence
520  vector<uint8_t> result;
521  for (size_t i = 0; i < blocks.at(0).size(); i++) {
522  for (size_t j = 0; j < blocks.size(); j++) {
523  // Skip the padding byte in short blocks
524  if (i != static_cast<unsigned int>(shortBlockLen - blockEccLen) || j >= static_cast<unsigned int>(numShortBlocks))
525  result.push_back(blocks.at(j).at(i));
526  }
527  }
528  if (result.size() != static_cast<unsigned int>(rawCodewords))
529  throw std::logic_error("Assertion error");
530  return result;
531  }

534  void QrCode::drawCodewords(const vector<uint8_t>& data) {
535  if (data.size() != static_cast<unsigned int>(getNumRawDataModules(version) / 8))
536  throw std::invalid_argument("Invalid argument");
537 
538  size_t i = 0; // Bit index into the data
539  // Do the funny zigzag scan
540  for (int right = size - 1; right >= 1; right -= 2) { // Index of right column in each column pair
541  if (right == 6)
542  right = 5;
543  for (int vert = 0; vert < size; vert++) { // Vertical counter
544  for (int j = 0; j < 2; j++) {
545  size_t x = static_cast<size_t>(right - j); // Actual x coordinate
546  bool upward = ((right + 1) & 2) == 0;
547  size_t y = static_cast<size_t>(upward ? size - 1 - vert : vert); // Actual y coordinate
548  if (!isFunction.at(y).at(x) && i < data.size() * 8) {
549  modules.at(y).at(x) = getBit(data.at(i >> 3), 7 - static_cast<int>(i & 7));
550  i++;
551  }
552  // If this QR Code has any remainder bits (0 to 7), they were assigned as
553  // 0/false/light by the constructor and are left unchanged by this method
554  }
555  }
556  }
557  if (i != data.size() * 8)
558  throw std::logic_error("Assertion error");
559  }

562  void QrCode::applyMask(int msk) {
563  if (msk < 0 || msk > 7)
564  throw std::domain_error("Mask value out of range");
565  size_t sz = static_cast<size_t>(size);
566  for (size_t y = 0; y < sz; y++) {
567  for (size_t x = 0; x < sz; x++) {
568  bool invert;
569  switch (msk) {
570  case 0: invert = (x + y) % 2 == 0; break;
571  case 1: invert = y % 2 == 0; break;
572  case 2: invert = x % 3 == 0; break;
573  case 3: invert = (x + y) % 3 == 0; break;
574  case 4: invert = (x / 3 + y / 2) % 2 == 0; break;
575  case 5: invert = x * y % 2 + x * y % 3 == 0; break;
576  case 6: invert = (x * y % 2 + x * y % 3) % 2 == 0; break;
577  case 7: invert = ((x + y) % 2 + x * y % 3) % 2 == 0; break;
578  default: throw std::logic_error("Assertion error");
579  }
580  modules.at(y).at(x) = modules.at(y).at(x) ^ (invert & !isFunction.at(y).at(x));
581  }
582  }
583  }

586  long QrCode::getPenaltyScore() const {
587  long result = 0;
588 
589  // Adjacent modules in row having same color, and finder-like patterns
590  for (int y = 0; y < size; y++) {
591  bool runColor = false;
592  int runX = 0;
593  std::array<int, 7> runHistory = {};
594  for (int x = 0; x < size; x++) {
595  if (module(x, y) == runColor) {
596  runX++;
597  if (runX == 5)
598  result += PENALTY_N1;
599  else if (runX > 5)
600  result++;
601  }
602  else {
603  finderPenaltyAddHistory(runX, runHistory);
604  if (!runColor)
605  result += finderPenaltyCountPatterns(runHistory) * PENALTY_N3;
606  runColor = module(x, y);
607  runX = 1;
608  }
609  }
610  result += finderPenaltyTerminateAndCount(runColor, runX, runHistory) * PENALTY_N3;
611  }
612  // Adjacent modules in column having same color, and finder-like patterns
613  for (int x = 0; x < size; x++) {
614  bool runColor = false;
615  int runY = 0;
616  std::array<int, 7> runHistory = {};
617  for (int y = 0; y < size; y++) {
618  if (module(x, y) == runColor) {
619  runY++;
620  if (runY == 5)
621  result += PENALTY_N1;
622  else if (runY > 5)
623  result++;
624  }
625  else {
626  finderPenaltyAddHistory(runY, runHistory);
627  if (!runColor)
628  result += finderPenaltyCountPatterns(runHistory) * PENALTY_N3;
629  runColor = module(x, y);
630  runY = 1;
631  }
632  }
633  result += finderPenaltyTerminateAndCount(runColor, runY, runHistory) * PENALTY_N3;
634  }
635 
636  // 2*2 blocks of modules having same color
637  for (int y = 0; y < size - 1; y++) {
638  for (int x = 0; x < size - 1; x++) {
639  bool color = module(x, y);
640  if (color == module(x + 1, y) &&
641  color == module(x, y + 1) &&
642  color == module(x + 1, y + 1))
643  result += PENALTY_N2;
644  }
645  }
646 
647  // Balance of dark and light modules
648  int dark = 0;
649  for (const vector<bool>& row : modules) {
650  for (bool color : row) {
651  if (color)
652  dark++;
653  }
654  }
655  int total = size * size; // Note that size is odd, so dark/total != 1/2
656  // Compute the smallest integer k >= 0 such that (45-5k)% <= dark/total <= (55+5k)%
657  int k = static_cast<int>((std::abs(dark * 20L - total * 10L) + total - 1) / total) - 1;
658  result += k * PENALTY_N4;
659  return result;
660  }

702  vector<uint8_t> QrCode::reedSolomonComputeDivisor(int degree) {
703  if (degree < 1 || degree > 255)
704  throw std::domain_error("Degree out of range");
705  // Polynomial coefficients are stored from highest to lowest power, excluding the leading term which is always 1.
706  // For example the polynomial x^3 + 255x^2 + 8x + 93 is stored as the uint8 array {255, 8, 93}.
707  vector<uint8_t> result(static_cast<size_t>(degree));
708  result.at(result.size() - 1) = 1; // Start off with the monomial x^0
709 
710  // Compute the product polynomial (x - r^0) * (x - r^1) * (x - r^2) * ... * (x - r^{degree-1}),
711  // and drop the highest monomial term which is always 1x^degree.
712  // Note that r = 0x02, which is a generator element of this field GF(2^8/0x11D).
713  uint8_t root = 1;
714  for (int i = 0; i < degree; i++) {
715  // Multiply the current product by (x - r^i)
716  for (size_t j = 0; j < result.size(); j++) {
717  result.at(j) = reedSolomonMultiply(result.at(j), root);
718  if (j + 1 < result.size())
719  result.at(j) ^= result.at(j + 1);
720  }
721  root = reedSolomonMultiply(root, 0x02);
722  }
723  return result;
724  }

727  vector<uint8_t> QrCode::reedSolomonComputeRemainder(const vector<uint8_t>& data, const vector<uint8_t>& divisor) {
728  vector<uint8_t> result(divisor.size());
729  for (uint8_t b : data) { // Polynomial division
730  uint8_t factor = b ^ result.at(0);
731  result.erase(result.begin());
732  result.push_back(0);
733  for (size_t i = 0; i < result.size(); i++)
734  result.at(i) ^= reedSolomonMultiply(divisor.at(i), factor);
735  }
736  return result;
737  }

740  uint8_t QrCode::reedSolomonMultiply(uint8_t x, uint8_t y) {
741  // Russian peasant multiplication
742  int z = 0;
743  for (int i = 7; i >= 0; i--) {
744  z = (z << 1) ^ ((z >> 7) * 0x11D);
745  z ^= ((y >> i) & 1) * x;
746  }
747  if (z >> 8 != 0)
748  throw std::logic_error("Assertion error");
749  return static_cast<uint8_t>(z);
750  }
